125            self, host: str, port: int, protocol: PROTOCOLS = TCP,

133        :param port: Subscription port.
134        :param protocol: Subscription protocol.

176    def _callback(self, data: bytes) -> bytes | None:
177        """
178        Internal callback for RPC calls.
179
180        :param data: Incoming data.
181        :return: Response data.
182        """
183        method, args, kwargs, = self._service.protocol.parse_call(data)
184        logger.info(self._service)
185        full_method_name = f"{self._service.__class__.__name__}.{method}"
186
187        logger.info(f"REGISTERED_METHODS: {_REGISTERED_METHODS}")
188        logger.info(f"full_method_name: {full_method_name}")
189
190        if full_method_name not in _REGISTERED_METHODS:
191            logger.info(
192                f"Method {full_method_name} is not registered in"
193                f" {_REGISTERED_METHODS}"
194            )
195            return None
196
197        _method = getattr(self._service, method)
198
199        try:
200            validate_or_ignore(_method, *args, **kwargs)
201            result = _method(*args, **kwargs)
202        except Exception as e:
203            error_callback = self._error_callback()
204            result = error_callback.handle_exception(e)
205
206        return self._service.protocol.build_response(result=result)
207
208    def _callback_event(self, topic: bytes, payload: bytes) -> None:
209        """
210        Internal callback for event subscription.
211
212        :param topic: Event topic.
213        :param payload: Event data.
214        """
215        service_name, topic = topic.decode().split(":")
216        key = (service_name, topic)
217
218        msg = self._service.protocol.parse_event(payload)
219
220        if key not in _REGISTERED_EVENT_HANDLERS.keys():
221            return None
222
223        _REGISTERED_EVENT_HANDLERS[key](self._service, msg)

236class ContainerRunner(Greenlet, metaclass=SingletonMeta):
237    """
238    Container runner for running the containers
239    """
240    LOOP_WAIT_TIME = 1
241    DELAY_BEFORE_START = 1
242
243    def __init__(
244            self, is_document_server: bool = True, host: str = "127.0.0.1",
245            port: int = 8081, output_dir: str = ".docs"
246    ) -> None:
247        Greenlet.__init__(self)
248        self._is_document_server = is_document_server
249        self._host = host
250        self._port = port
251        self._output_dir = output_dir
252        self._containers: dict[str, dict[str, Any]] = {}
253        self._workers: list[Greenlet] = []
254
255    def register(
256            self, name: str, container: Container,
257            *args: List[Any], **kwargs: Any
258    ) -> None:
259        """
260        Register container
261        """
262        self._containers[name] = {
263            "container": container,
264            "args": args,
265            "kwargs": kwargs
266        }
267
268    def _run(self) -> None:
269        """
270        Run all registered containers
271        """
272
273        docs_paths: list[str | Path] = []
274
275        for name, container in self._containers.items():
276            worker = Greenlet(
277                run=container["container"].run,
278                *container["args"], **container["kwargs"]
279            )
280            worker.start()
281            self._workers.append(worker)
282
283            if self._is_document_server:
284                docs_paths.extend(
285                    get_paths(
286                        container["container"]._service_class.__name__  # type: ignore  # noqa
287                    )
288                )
289
290        if self._is_document_server:
291            thread = Greenlet(
292                generate_docs_for_service,
293                docs_paths, output_dir=self._output_dir
294            )
295            thread.start()
296
297            doc_server = Greenlet(
298                run=start_docs_server,
299                **dict(modules=docs_paths, host=self._host, port=self._port)
300            )
301            logger.info(f"Starting docs server on {self._host}:{self._port}")
302            doc_server.start()
303
304        while True:
305            gevent.sleep(self.LOOP_WAIT_TIME)
306            for worker in self._workers:
307                if worker.dead and not worker.successful():
308                    self._clear_worker(worker)
309                    self._restart_worker(worker)
310
311            if not self._workers:
312                break
313
314    def start(self) -> None:
315        """
316        Start the containers
317        """
318        self._run()
319
320    def _clear_worker(self, worker: Greenlet) -> None:
321        """
322        Clear the worker
323        """
324        worker.kill()
325        self._workers.remove(worker)
326        logger.info(f"Worker {worker} is dead")
327
328    def _restart_worker(self, worker: Greenlet) -> None:
329        """
330        Restart the worker
331        """
332        worker = Greenlet(
333            run=worker.run,
334            *worker.args, **worker.kwargs
335        )
336        worker.start()
337        self._workers.append(worker)
338        logger.info(f"Worker {worker} is restarted")

Changed in version 1.1b1: The run argument to the constructor is now verified to be a callable object. Previously, passing a non-callable object would fail after the greenlet was spawned.

Changed in version 1.3b1: The GEVENT_TRACK_GREENLET_TREE configuration value may be set to a false value to disable spawn_tree_locals, spawning_greenlet, and spawning_stack. The first two will be None in that case, and the latter will be empty.

Changed in version 1.5: Greenlet objects are now more careful to verify that their parent is really a gevent hub, raising a TypeError earlier instead of an AttributeError later.

Changed in version 20.12.1: Greenlet objects now function as context managers. Exiting the with suite ensures that the greenlet has completed by joining <join>() the greenlet (blocking, with no timeout). If the body of the suite raises an exception, the greenlet is killed <kill>() with the default arguments and not joined in that case.
